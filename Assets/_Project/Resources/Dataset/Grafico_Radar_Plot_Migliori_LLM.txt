import numpy as np
import matplotlib.pyplot as plt

# Dati
labels = ['Acc', 'Coh', 'Comp', 'Nat']
models = {
    'Qwen2.5-coder-14b': [4.75, 4.88, 4.38, 4.62],
    'Gpt-oss-20b': [4.62, 4.75, 3.88, 4.62],
    'Gemma-3-27b-it': [4.88, 4.75, 4.62, 4.75]
}

angles = np.linspace(0, 2 * np.pi, len(labels), endpoint=False)
angles = np.concatenate((angles, [angles[0]]))  # chiudi il poligono

plt.figure(figsize=(6,6))

for model, values in models.items():
    values = np.concatenate((values, [values[0]]))
    plt.polar(angles, values, marker='o', label=model)

plt.xticks(angles[:-1], labels)
plt.ylim(0, 5)
plt.title("Confronto LLM sulle metriche")
plt.legend(loc='upper right', bbox_to_anchor=(1.3, 1.1))
plt.tight_layout()
plt.show()
